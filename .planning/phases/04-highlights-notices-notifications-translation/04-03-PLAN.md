**1. notify Edge Function** — Create `packages/supabase/functions/notify/index.ts`:

Follow the `generate-nickname` Deno.serve pattern. The function:
- Accepts POST with JSON body: `{ event_type, community_id, data: { title, body, deep_link, actor_id? } }`
- Creates supabaseAdmin client with `SUPABASE_SERVICE_ROLE_KEY`
- Maps `event_type` to notification_preferences column:
  - `creator_post` → `creator_posts`
  - `comment` → `comments`
  - `like` → `likes`
  - `notice` → `notices`
  - `member_post` → `creator_posts` (uses same pref as creator posts)
- Fetches target users who are members of the community AND have the relevant preference enabled (or no preference row = defaults to true):
  ```sql
  SELECT DISTINCT cm.user_id, pt.token
  FROM community_members cm
  LEFT JOIN notification_preferences np
    ON np.user_id = cm.user_id AND np.community_id = cm.community_id
  LEFT JOIN push_tokens pt ON pt.user_id = cm.user_id
  WHERE cm.community_id = $community_id
    AND (np.{pref_column} IS NULL OR np.{pref_column} = true)
    AND cm.user_id != $actor_id  -- don't notify the actor
  ```
  Use supabaseAdmin.rpc or raw SQL via supabaseAdmin.from('community_members')... The simplest approach: use a database function `get_notify_targets(p_community_id uuid, p_event_type text, p_actor_id uuid)` created in the migration, or do it with multiple queries.

  Simplified approach without RPC:
  ```typescript
  // Get community members
  const { data: members } = await supabaseAdmin
    .from('community_members')
    .select('user_id')
    .eq('community_id', community_id);

  // Get preferences that explicitly disabled this type
  const prefColumn = prefMap[event_type]; // e.g., 'creator_posts'
  const { data: disabledPrefs } = await supabaseAdmin
    .from('notification_preferences')
    .select('user_id')
    .eq('community_id', community_id)
    .eq(prefColumn, false);

  const disabledUserIds = new Set((disabledPrefs ?? []).map(p => p.user_id));
  const actorId = data.actor_id;
  const targetUserIds = (members ?? [])
    .map(m => m.user_id)
    .filter(uid => uid !== actorId && !disabledUserIds.has(uid));

  // Get push tokens for targets
  const { data: tokens } = await supabaseAdmin
    .from('push_tokens')
    .select('user_id, token')
    .in('user_id', targetUserIds);
  ```

- For `member_post` event: additionally filter to only users who follow the specific member. Query `community_follows` table: `SELECT follower_id FROM community_follows WHERE following_id = $member_user_id AND community_id = $community_id`. Intersect with targetUserIds.

- Bulk insert in-app notifications:
  ```typescript
  await supabaseAdmin.from('notifications').insert(
    targetUserIds.map(userId => ({
      user_id: userId,
      community_id: community_id,
      type: event_type,
      title: data.title,
      body: data.body,
      data: { ...data.deep_link },
    }))
  );
  ```

- Send Expo Push notifications (only for users with tokens):
  ```typescript
  const pushTokens = (tokens ?? []).map(t => t.token).filter(Boolean);
  if (pushTokens.length > 0) {
    const messages = pushTokens.map(to => ({
      to,
      title: data.title,
      body: data.body,
      data: data.deep_link,
      sound: 'default' as const,
    }));
    // Batch in groups of 100 (Expo Push API limit)
    for (let i = 0; i < messages.length; i += 100) {
      const batch = messages.slice(i, i + 100);
      await fetch('https://exp.host/--/api/v2/push/send', {
        method: 'POST',
        headers: { 'Accept': 'application/json', 'Content-Type': 'application/json' },
        body: JSON.stringify(batch),
      });
    }
  }
  ```

- Return `{ notified: targetUserIds.length, pushed: pushTokens.length }`

-- Helper: get supabase functions URL for local/cloud
-- For local dev, net.http_post uses localhost:54321/functions/v1/notify
-- For cloud, use the project's functions URL

-- Trigger for creator posts (also covers member_post via community_follows)
CREATE OR REPLACE FUNCTION trigger_notify_on_creator_post()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.author_role = 'creator' THEN
    PERFORM net.http_post(
      url := COALESCE(
        current_setting('app.supabase_functions_url', true),
        'http://localhost:54321/functions/v1'
      ) || '/notify',
      headers := jsonb_build_object(
        'Authorization', 'Bearer ' || COALESCE(
          current_setting('app.service_role_key', true),
          ''
        ),
        'Content-Type', 'application/json'
      ),
      body := jsonb_build_object(
        'event_type', 'creator_post',
        'community_id', NEW.community_id::text,
        'data', jsonb_build_object(
          'title', 'New creator post',
          'body', LEFT(NEW.content, 100),
          'actor_id', NEW.user_id::text,
          'deep_link', jsonb_build_object('post_id', NEW.id::text, 'community_id', NEW.community_id::text)
        )
      )
    );
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER posts_creator_notify_trigger
  AFTER INSERT ON posts
  FOR EACH ROW
  EXECUTE FUNCTION trigger_notify_on_creator_post();

-- Trigger for comments on posts (notify post author)
CREATE OR REPLACE FUNCTION trigger_notify_on_comment()
RETURNS TRIGGER AS $$
DECLARE
  v_post_author_id uuid;
  v_post_community_id uuid;
  v_post_content text;
BEGIN
  SELECT user_id, community_id, LEFT(content, 50) INTO v_post_author_id, v_post_community_id, v_post_content
  FROM posts WHERE id = NEW.post_id;

  IF v_post_author_id IS NOT NULL AND v_post_author_id != NEW.user_id THEN
    PERFORM net.http_post(
      url := COALESCE(current_setting('app.supabase_functions_url', true), 'http://localhost:54321/functions/v1') || '/notify',
      headers := jsonb_build_object(
        'Authorization', 'Bearer ' || COALESCE(current_setting('app.service_role_key', true), ''),
        'Content-Type', 'application/json'
      ),
      body := jsonb_build_object(
        'event_type', 'comment',
        'community_id', v_post_community_id::text,
        'data', jsonb_build_object(
          'title', 'New comment',
          'body', LEFT(NEW.content, 100),
          'actor_id', NEW.user_id::text,
          'deep_link', jsonb_build_object('post_id', NEW.post_id::text, 'community_id', v_post_community_id::text)
        )
      )
    );
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER comments_notify_trigger
  AFTER INSERT ON comments
  FOR EACH ROW
  EXECUTE FUNCTION trigger_notify_on_comment();

-- Trigger for likes on posts (notify post author)
CREATE OR REPLACE FUNCTION trigger_notify_on_like()
RETURNS TRIGGER AS $$
DECLARE
  v_post_author_id uuid;
  v_post_community_id uuid;
BEGIN
  IF NEW.target_type = 'post' THEN
    SELECT user_id, community_id INTO v_post_author_id, v_post_community_id
    FROM posts WHERE id = NEW.target_id;

    IF v_post_author_id IS NOT NULL AND v_post_author_id != NEW.user_id THEN
      PERFORM net.http_post(
        url := COALESCE(current_setting('app.supabase_functions_url', true), 'http://localhost:54321/functions/v1') || '/notify',
        headers := jsonb_build_object(
          'Authorization', 'Bearer ' || COALESCE(current_setting('app.service_role_key', true), ''),

        ),
        body := jsonb_build_object(
          'event_type', 'like',
          'community_id', v_post_community_id::text,
          'data', jsonb_build_object(
            'title', 'Post liked',
            'body', '',
            'actor_id', NEW.user_id::text,
            'deep_link', jsonb_build_object('post_id', NEW.target_id::text, 'community_id', v_post_community_id::text)

        )
      );
    END IF;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

**3. Push token registration hook** — `apps/mobile/hooks/notification/usePushTokenRegistration.ts`:
```typescript
import * as Notifications from 'expo-notifications';
import * as Device from 'expo-device';
import { Platform } from 'react-native';
import { useEffect } from 'react';
import { supabase } from '../../lib/supabase';
import { useAuthStore } from '../../stores/authStore';

export function usePushTokenRegistration() {
  const user = useAuthStore((s) => s.user);

  useEffect(() => {
    if (!user?.id) return;
    registerForPushNotificationsAsync(user.id);
  }, [user?.id]);
}

async function registerForPushNotificationsAsync(userId: string) {
  if (!Device.isDevice) return; // Must be physical device

  const { status: existingStatus } = await Notifications.getPermissionsAsync();
  let finalStatus = existingStatus;

  if (existingStatus !== 'granted') {
    const { status } = await Notifications.requestPermissionsAsync();
    finalStatus = status;
  }

  if (finalStatus !== 'granted') return;

  if (Platform.OS === 'android') {
    await Notifications.setNotificationChannelAsync('default', {
      name: 'default',
      importance: Notifications.AndroidImportance.MAX,
    });
  }

  const tokenData = await Notifications.getExpoPushTokenAsync({
    projectId: process.env.EXPO_PUBLIC_EAS_PROJECT_ID,
  });

  await supabase.from('push_tokens').upsert({
    user_id: userId,
    token: tokenData.data,
    platform: Platform.OS,
    updated_at: new Date().toISOString(),
  }, { onConflict: 'user_id' });
}
```

**1. useNotifications hook** — `apps/mobile/hooks/notification/useNotifications.ts`:
```typescript
import { useQuery } from '@tanstack/react-query';
import { supabase } from '../../lib/supabase';

export interface Notification {
  id: string;
  type: 'creator_post' | 'comment' | 'like' | 'notice' | 'member_post' | 'system';
  title: string;
  body: string;
  data: Record<string, string> | null;
  is_read: boolean;
  created_at: string;
}

export function useNotifications(communityId: string) {
  return useQuery<Notification[]>({
    queryKey: ['notifications', communityId],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('notifications')
        .select('id, type, title, body, data, is_read, created_at')
        .eq('community_id', communityId)
        .order('created_at', { ascending: false })
        .limit(50);
      if (error) throw error;
      return data as Notification[];
    },
    enabled: !!communityId,
  });
}
```
