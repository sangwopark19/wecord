    **Step 6: Create posts_with_nickname view using Drizzle pgView:**
    Add to packages/db/src/schema/content.ts:
    ```typescript
    export const postsWithNickname = pgView('posts_with_nickname').as((qb) =>
      qb.select({
        // all posts columns
        ...posts,
        authorNickname: communityMembers.communityNickname,
        authorCmId: communityMembers.id,
        artistMemberName: artistMembers.displayName,
        communityName: communities.name,
        communitySlug: communities.slug,
      })
      .from(posts)
      .innerJoin(communityMembers, and(
        eq(communityMembers.userId, posts.authorId),
        eq(communityMembers.communityId, posts.communityId)
      ))
      .leftJoin(artistMembers, eq(artistMembers.id, posts.artistMemberId))
      .innerJoin(communities, eq(communities.id, posts.communityId))
    );
    ```
