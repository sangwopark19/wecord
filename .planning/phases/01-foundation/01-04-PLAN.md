    **Step 3: Create .github/workflows/ci.yml:**
    Per user decision (CONTEXT.md): "GitHub Actions: Lint + TypeCheck, Build, Supabase migration test (Docker), EAS Build trigger"

    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      lint-typecheck:
        name: Lint & Typecheck
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4

          - uses: pnpm/action-setup@v4
            with:
              version: 9

          - uses: actions/setup-node@v4
            with:
              node-version: '22'
              cache: 'pnpm'

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Lint
            run: pnpm turbo lint

          - name: Typecheck
            run: pnpm turbo typecheck

      build:
        name: Build
        runs-on: ubuntu-latest
        needs: lint-typecheck
        steps:
          - uses: actions/checkout@v4

          - uses: pnpm/action-setup@v4
            with:
              version: 9

          - uses: actions/setup-node@v4
            with:
              node-version: '22'
              cache: 'pnpm'

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Build all packages
            run: pnpm turbo build

      migration-test:
        name: Supabase Migration Test
        runs-on: ubuntu-latest
        needs: lint-typecheck
        steps:
          - uses: actions/checkout@v4

          - uses: supabase/setup-cli@v1
            with:
              version: latest

          - name: Start Supabase local
            run: |
              cd packages/supabase
              supabase start -x realtime,storage-api,imgproxy,inbucket,postgrest,gotrue,studio,edge-runtime

          - name: Apply migrations
            run: |
              cd packages/supabase
              supabase db push --local

          - name: Verify tables exist
            run: |
              cd packages/supabase
              supabase db execute "SELECT count(*) as table_count FROM pg_tables WHERE schemaname = 'public';" --local

          - name: Verify RLS enabled
            run: |
              cd packages/supabase
              supabase db execute "SELECT count(*) as rls_count FROM pg_tables WHERE schemaname = 'public' AND rowsecurity = true;" --local

          - name: Stop Supabase
            if: always()
            run: |
              cd packages/supabase
              supabase stop

      test:
        name: Test
        runs-on: ubuntu-latest
        needs: lint-typecheck
        steps:
          - uses: actions/checkout@v4

          - uses: pnpm/action-setup@v4
            with:
              version: 9

          - uses: actions/setup-node@v4
            with:
              node-version: '22'
              cache: 'pnpm'

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Run tests
            run: pnpm turbo test

      eas-build:
        name: EAS Build
        if: startsWith(github.ref, 'refs/tags/v')
        runs-on: ubuntu-latest
        needs: [build, test, migration-test]
        steps:
          - uses: actions/checkout@v4

          - uses: pnpm/action-setup@v4
            with:
              version: 9

          - uses: actions/setup-node@v4
            with:
              node-version: '22'
              cache: 'pnpm'

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Setup EAS
            uses: expo/expo-github-action@v8
            with:
              eas-version: latest
              token: ${{ secrets.EXPO_TOKEN }}

          - name: Build
            working-directory: apps/mobile
            run: eas build --platform all --non-interactive --profile production
    ```

    Key design decisions per CONTEXT.md:
    - lint-typecheck runs on every push to main and every PR
    - build, test, and migration-test run after lint-typecheck passes
    - migration-test uses supabase/setup-cli action to start a local Supabase Docker instance, applies migrations, and verifies table/RLS counts
    - EAS Build ONLY triggers on release tags (v*.*.*) to save build costs
    - EAS Build depends on all other jobs (build, test, migration-test)
