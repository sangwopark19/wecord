```typescript
import 'react-native-url-polyfill/auto';
import { createClient } from '@supabase/supabase-js';
import * as SecureStore from 'expo-secure-store';

const ExpoSecureStoreAdapter = {
  getItem: (key: string) => SecureStore.getItemAsync(key),
  setItem: (key: string, value: string) => SecureStore.setItemAsync(key, value),
  removeItem: (key: string) => SecureStore.deleteItemAsync(key),
};

export const supabase = createClient(
  process.env.EXPO_PUBLIC_SUPABASE_URL!,
  process.env.EXPO_PUBLIC_SUPABASE_ANON_KEY!,
  {
    auth: {
      storage: ExpoSecureStoreAdapter,
      autoRefreshToken: true,
      persistSession: true,
      detectSessionInUrl: false, // MUST be false in React Native
    },
  }
);
```

**5. Create apps/mobile/stores/authStore.ts:**
Zustand store with the following shape:
- `session: Session | null`
- `user: User | null`
- `profile: { userId: string; globalNickname: string; avatarUrl: string | null; language: 'ko'|'en'|'th'|'zh'|'ja'; onboardingCompleted: boolean; dateOfBirth: string | null } | null`
- `loading: boolean`
- `setSession(session)` — sets session + user
- `setProfile(profile)` — sets profile
- `initialize()` — calls `supabase.auth.getSession()` to restore from SecureStore, then subscribes to `onAuthStateChange`. On `SIGNED_IN` event, fetch profile from `profiles` table. If no profile exists (new user), call `generate-nickname` Edge Function, upsert profile with `onboarding_completed: false`, default language from device locale (expo-localization `getLocales()[0]?.languageCode` mapped to supported language or 'en' fallback). Set `loading: false` after init.
- `signOut()` — calls `supabase.auth.signOut()`, clears session/user/profile

**6. Create apps/mobile/hooks/useAuth.ts:**
Thin hook that re-exports `useAuthStore` selectors for convenience:
- `useAuth()` returns `{ session, user, profile, loading, signOut }`

**7. Create packages/supabase/functions/generate-nickname/index.ts:**
Deno Edge Function that:
- Reads `Authorization` header from request
- Creates admin client using `SUPABASE_URL` + `SUPABASE_SERVICE_ROLE_KEY` env vars
- Generates `User#XXXX` (random 4-digit 1000-9999)
- Checks uniqueness against `profiles.global_nickname` (retry up to 10 times)
- Returns JSON `{ nickname: "User#XXXX" }`
- Uses `Deno.serve()` (current Supabase Edge Function pattern, NOT deprecated `serve` from std/http)

```typescript
    const { data, error } = await supabase.auth.signInWithOAuth({

      options: { redirectTo, skipBrowserRedirect: true },

    if (error) throw error;
    if (data.url) {
      const result = await WebBrowser.openAuthSessionAsync(data.url, redirectTo);
      if (result.type === 'success') {
        await supabase.auth.exchangeCodeForSession(result.url);
      }
    }
  } catch (err) {
    // Show error toast
  } finally {
    setLoading(false);
  }
}
```
